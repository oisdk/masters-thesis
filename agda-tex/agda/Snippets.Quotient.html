data _/_ (A : Type a) (R : A → A → Type b) : Type (a ℓ⊔ b) where
  [_] : (x : A) → A / R
  eq/ : (x y : A) → (r : R x y) → [ x ] ≡ [ y ]
  squash/ : (x y : A / R) → (p q : x ≡ y) → p ≡ q
